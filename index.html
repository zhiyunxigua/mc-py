3 在main.py注册这个模块

导入ExampleModule.py这个模块的类

创建一个全局变量ExampleModule

给全局变量赋值ExampleModule()

Python 
# coding=utf-8
from utils import *
from module.ExampleModule import ExampleModule  # 导入这个模块的类  <------


class Main(BaseServer):
    def __init__(self, namespace, systemName):
        super(Main, self).__init__(namespace, systemName)
        self.ExampleModule = None  # 创建一个全局变量    <------------

    @Listen()
    def LoadServerAddonScriptsAfter(self, args):
        self.register_module()

    def register_module(self):
        self.ExampleModule = ExampleModule()  # 赋值    <-----------

监听器教程
1 什么都不传入

@Listen()

Python 
@Listen()
def 事件名():
    pass
2 可以传入这三个参数

@Listen("事件名", 系统名, 优先级)

优先级越高越后处理，默认3

Python 
@Listen("eventName", Listen.server, 0)
def listen_example():
    pass
3 可以只传入指定参数

@Listen(event_type="系统名")

Python 
@Listen(event_name=事件名)
def listen_example():
    pass

@Listen(event_type=example)
def 事件名():
    pass

@Listen(priority=5)
def 事件名():
    pass
